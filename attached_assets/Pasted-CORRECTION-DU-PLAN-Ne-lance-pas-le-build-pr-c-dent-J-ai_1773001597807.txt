export const scrapeGoogleSearchTool = createTool({
  id: "scrapeGoogleSearch-serp",
  description: "Search for eco-focused architecture firms using SerpAPI",
  inputSchema: z.object({
    placesCount: z.number().optional(),
  }),
  outputSchema: z.object({
    firms: z.array(z.any()),
    placesCount: z.number(),
    searchCount: z.number(),
  }),
  execute: async ({ context }) => {
    const placesCount = context?.placesCount || 0;
    const firms: any[] = [];

    const ecoQueries = [
      'architecte "BTC" OR "brique de terre comprimée" Sénégal',
      'cabinet architecture "pisé" OR "terre crue" Sénégal',
      'construction "matériaux locaux" OR "bio-sourcés" Sénégal',
      '"architecture bioclimatique" OR "éco-construction" Sénégal',
      '"CRAterre" OR "architecture en terre" Sénégal',
    ];

    for (const query of ecoQueries) {
      try {
        const serpUrl = `https://serpapi.com/search.json?q=${encodeURIComponent(query)}&api_key=${process.env.SERPAPI}&num=10`;
        
        const response = await fetch(serpUrl);
        const data = await response.json();

        for (const result of data.organic_results || []) {
          try {
            const url = new URL(result.link);
            const domain = url.hostname.replace('www.', '');

            // Ignorer agrégateurs
            const excludeDomains = ['wikipedia', 'archdaily', 'pinterest', 'facebook', 'linkedin'];
            if (excludeDomains.some(d => domain.includes(d))) continue;

            firms.push({
              name: result.title,
              country: "Sénégal",
              city: null,
              address: null,
              website: result.link,
              phone: null,
              source: "SerpAPI Google Search",
              notes: result.snippet || "",
              discoveredAt: new Date().toISOString(),
              status: "new",
            });
          } catch (e) {
            // URL invalide, skip
          }
        }

        await new Promise(r => setTimeout(r, 1000)); // 1s pause entre requêtes
      } catch (error) {
        console.error(`SerpAPI error for query "${query}":`, error);
      }
    }

    // Déduplication par domaine
    const seen = new Set();
    const uniqueFirms = firms.filter(f => {
      if (!f.website) return false;
      const domain = new URL(f.website).hostname;
      if (seen.has(domain)) return false;
      seen.add(domain);
      return true;
    });

    console.log(`SerpAPI: Found ${uniqueFirms.length} unique firms`);

    return {
      firms: uniqueFirms,
      placesCount,
      searchCount: uniqueFirms.length,
    };
  },
});
