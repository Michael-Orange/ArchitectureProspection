Dans `scrapeGoogleSearchSerpAPITool`, ajoute un filtre intelligent pour garder UNIQUEMENT les sites de cabinets :

```typescript
const excludePatterns = [
  // Sous-pages d'articles
  '/la-construction/',
  '/focus/',
  '/bonne-pratique/',
  '/case/',
  '/programme-de-construction',
  '/darchitecture-en-terre-crue',  // Page blog
  
  // Agrégateurs/médias à exclure complètement
  'climate-chance.org',
  'afrik21.africa',
  'craterre.org',  // On les scrapera via Track C
];

const validDomainPatterns = [
  // Formats de sites de cabinets valides
  /^https?:\/\/[a-z0-9-]+\.(com|sn|net|org)\/?$/i,  // Homepage (ex: worofila.com/)
  /^https?:\/\/www\.[a-z0-9-]+\.(com|sn|net|org)\/?$/i,  // Homepage www
  /^https?:\/\/[a-z0-9-]+\.(com|sn|net|org)\/atelier\/?$/i,  // /atelier (ok)
  /^https?:\/\/[a-z0-9-]+\.(com|sn|net|org)\/presentation\/?$/i,  // /presentation (ok)
];

for (const result of data.organic_results || []) {
  const url = result.link;
  
  // Exclure les patterns interdits
  if (excludePatterns.some(pattern => url.toLowerCase().includes(pattern))) {
    console.log(`Filtered out article: ${result.title}`);
    continue;
  }
  
  // Garder uniquement les formats valides (homepage ou pages institutionnelles)
  const isValidFormat = validDomainPatterns.some(regex => regex.test(url));
  
  if (isValidFormat) {
    // Extraire le domaine propre
    const domain = new URL(url).hostname.replace('www.', '');
    
    firms.push({
      name: result.title.replace(/ - .+$/, '').trim(),  // Nettoyer le titre
      website: url,
      domain: domain,
      snippet: result.snippet,
      source: "SerpAPI Google Search",
    });
  }
}
```
